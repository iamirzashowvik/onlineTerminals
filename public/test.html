      int num1, num2, sum;

      cin >> num1;

C

To use scanf please use "fflush(stdout);" before scanf
      #include <stdio.h>
      
      int main() {
          int num1, num2, sum;
          
          printf("Enter two integers: ");
          fflush(stdout); // Force stdout to be flushed
      
          // Read the input from the user and store them in variables
          scanf("%d %d", &num1, &num2);
      
          // Calculate the sum
          sum = num1 + num2;
      
          // Print the result
          printf("The sum of %d and %d is %d\n", num1, num2, sum);
      
          return 0;
      }
